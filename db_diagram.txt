enum FlightStatus {
  SCHEDULED
  DEPARTED
  CANCELLED
  COMPLETED
}

Table Flight {
  id integer [primary key]
  flight_number varchar(7)
  flight_date date
  company text
  origin varchar(3)
  destination varchar(3)
  departure_datetime timestamp
  arrival_datetime timestamp

  seat_capacity integer
  seat_available integer

  price integer
  
  status FlightStatus
  
  checks {
    `seat_available >= 0`
    `seat_capacity > 0`
    `price > 0`
  }

  indexes {
    (flight_number, flight_date) [unique]
  }
}

enum BookingStatus {
  CONFIRMED
  CANCELLED
}

Table Booking {
  id integer [primary key]
  flight_id integer
  passenger_name varchar(64)
  passenger_emal varchar(64)
  seat_count integer
  price integer
  status BookingStatus

  checks {
    `seat_count > 0`
    `price > 0`
  }
}

enum SeatReservationStatus {
  ACTIVE
  RELEASED
  EXPIRED
}

Table SeatReservation {
  booking_id integer [primary key]
  flight_id integer
  seat_count integer
  status SeatReservationStatus
  
  checks {
    `seat_count > 0`
  }
}

Ref: Booking.flight_id > Flight.id
Ref: SeatReservation.flight_id > Flight.id
Ref: SeatReservation.booking_id > Booking.id
